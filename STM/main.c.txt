void delay (uint16_t time){
	__HAL_TIM_SET_COUNTER(&htim4,0);
	while (__HAL_TIM_GET_COUNTER(&htim1)<time);

void HAL_TIM_IC_CaptureCallback(TIM_HandleTypeDef *htim) {//timer 4 channel 1
	if (htim->Channel == HAL_TIM_ACTIVE_CHANNEL_1) // if the interrupt source is channel1
			{
		if (Is_First_Captured == 0) // if the first value is not captured
				{
			IC_Val1 = HAL_TIM_ReadCapturedValue(htim, TIM_CHANNEL_1); // read the first value
			Is_First_Captured = 1;  // set the first captured as true
			// Now change the polarity to falling edge
			__HAL_TIM_SET_CAPTUREPOLARITY(htim, TIM_CHANNEL_1,
					TIM_INPUTCHANNELPOLARITY_FALLING);
		}

		else if (Is_First_Captured == 1)   // if the first is already captured
				{
			IC_Val2 = HAL_TIM_ReadCapturedValue(htim, TIM_CHANNEL_1); // read second value
			__HAL_TIM_SET_COUNTER(htim, 0);  // reset the counter

			if (IC_Val2 > IC_Val1) {
				Difference = IC_Val2 - IC_Val1;
			}

			else if (IC_Val1 > IC_Val2) {
				Difference = (0xffff - IC_Val1) + IC_Val2;
			}

			Distance = Difference * .034 / 2;
			Is_First_Captured = 0; // set it back to false

			// set polarity to rising edge
			__HAL_TIM_SET_CAPTUREPOLARITY(htim, TIM_CHANNEL_1,
					TIM_INPUTCHANNELPOLARITY_RISING);
			__HAL_TIM_DISABLE_IT(&htim4, TIM_IT_CC1);
		}
	}
}

				choice = 'W'; // move slightly forward after first turn
				value = 10;
				osDelay(200);
				while(current!='X'){}//wait for movement to finish
				osDelay(100);
				choice='C';//CREEP IN
				value=150;
				osDelay(200);
				while(current!='X'){}//wait for movement to finish
				osDelay(100);
				choice='W';//MOVE IN parking lot
				value=10;
				osDelay(200);
			}
			customflag=0;
			HAL_UART_Transmit(&huart3, (uint8_t*) &customend, 1, 0xFFFFFFFF);//SEND K
		}
		osDelay(1);
	}
  /* USER CODE END custom */
}
